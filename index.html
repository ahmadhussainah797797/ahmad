function shrinkNo() {
  const noBtn = document.getElementById("no");

    yesSize += 4;

    noBtn.style.fontSize = noSize + "px";
    document.getElementById("yes").style.fontSize = yesSize + "px";
  }

  noBtn.style.left = Math.random() * 80 + "vw";
  noBtn.style.top = Math.random() * 80 + "vh";
}
